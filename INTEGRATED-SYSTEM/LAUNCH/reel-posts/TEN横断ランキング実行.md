### **STEP3: TENスコア統合算出**
```javascript
// 統合スコア算出ロジック
function calculateIntegratedTENScore(items) {
  return items.map(item => {
    const baseScore = item.evaluation || {};
    
    // DB種別によるボーナス
    const dbBonus = getDBCategoryBonus(item.dbType);
    
    // 相乗効果ボーナス（組み合わせ）
    const synergyBonus = calculateSynergyBonus(item, otherItems);
    
    // 最終TENスコア
    const finalScore = 
      baseScore.immediacy * 0.20 +
      baseScore.simplicity * 0.20 +
      baseScore.popularity * 0.15 +
      baseScore.costPerformance * 0.15 +
      baseScore.specialization * 0.20 +
      baseScore.productivityGain * 0.10 +
      dbBonus +
      synergyBonus;
      
    return {
      ...item,
      tenScore: Math.round(finalScore * 10) / 10,
      grade: getTENGrade(finalScore)
    };
  }).sort((a, b) => b.tenScore - a.tenScore);
}
```
